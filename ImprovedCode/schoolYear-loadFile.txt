void schoolYear::loadFile()
{
	if (pHeadSchoolYear)
		return; // LL already loaded

	std::ifstream fin("../Data/SchoolYear/all school year.txt");
	if (!fin.is_open())
	{
		std::cerr << "Failed to open the file.\n";
		return;
	}

	std::string year;
	while (getline(fin, year))
	{
		if (year.empty())
			break;

		schoolYear* tmp = new schoolYear(year, nullptr);
		if (!pHeadSchoolYear)
		{
			pHeadSchoolYear = tmp;
			pTailSchoolYear = tmp;
		}
		else
		{
			pTailSchoolYear->pNext = tmp;
			pTailSchoolYear = tmp;
		}
	}

	curSchoolYear = pHeadSchoolYear;
	fin.close();
}
