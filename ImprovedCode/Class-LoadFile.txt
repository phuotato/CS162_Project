void Class::LoadFile()
{
    if (pHeadClass) return; // If the linked list is already populated, return

    std::ifstream fin("../Data/Class/Class.txt");

    std::string curName;
    while (getline(fin, curName))
    {
        if (curName.empty())
            continue;

        if (!pHeadClass)
        {
            pHeadClass = new Class(curName);
            pTailClass = pHeadClass;
        }
        else
        {
            Class* tmp = new Class(curName);
            pTailClass->pNext = tmp;
            pTailClass = tmp;
        }
    }

    fin.close();
}
